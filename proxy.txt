import { NextResponse } from "next/server";
import type { NextRequest } from "next/server";
import { createServerClient } from "@supabase/ssr";
import { cookies } from "next/headers";

// Environment variable validation
function validateEnv() {
  const required = ["NEXT_PUBLIC_SUPABASE_URL", "NEXT_PUBLIC_SUPABASE_ANON_KEY"];
  const missing = required.filter((key) => !process.env[key]);
  
  if (missing.length > 0) {
    throw new Error(
      `Missing required environment variables: ${missing.join(", ")}`
    );
  }
}

// Create Supabase client
async function createSupabaseClient(cookieStore: Awaited<ReturnType<typeof cookies>>) {
  return createServerClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!,
    {
      cookies: {
        getAll() {
          return cookieStore.getAll();
        },
        setAll(cookiesToSet) {
          try {
            cookiesToSet.forEach(({ name, value, options }) =>
              cookieStore.set(name, value, options)
            );
          } catch {
            // Silently fail - cookies can't be set in middleware response
          }
        },
      },
    }
  );
}

// Check if route is protected
function isProtectedRoute(pathname: string): boolean {
  return pathname.startsWith("/dashboard");
}

// Check if route is public auth-related
function isPublicAuthRoute(pathname: string): boolean {
  return ["/login", "/register", "/auth/callback"].includes(pathname);
}

export async function proxy(request: NextRequest) {
  try {
    validateEnv();
  } catch (error) {
    console.error("[Auth Middleware]", error);
    return NextResponse.redirect(new URL("/login", request.url));
  }

  const { pathname } = request.nextUrl;
  const cookieStore = await cookies();
  
  try {
    const supabase = await createSupabaseClient(cookieStore);
    const {
      data: { session },
    } = await supabase.auth.getSession();

    const hasSession = !!session;

    // Redirect unauthenticated users away from protected routes
    if (isProtectedRoute(pathname) && !hasSession) {
      const loginUrl = new URL("/login", request.url);
      loginUrl.searchParams.set("redirect", pathname);
      return NextResponse.redirect(loginUrl);
    }

    // Redirect authenticated users away from auth routes (login, register)
    if (isPublicAuthRoute(pathname) && hasSession) {
      return NextResponse.redirect(new URL("/dashboard", request.url));
    }

    return NextResponse.next();
  } catch (error) {
    console.error("[Auth Middleware] Session check failed:", error);
    
    // If protected route and session check fails, redirect to login
    if (isProtectedRoute(pathname)) {
      const loginUrl = new URL("/login", request.url);
      loginUrl.searchParams.set("redirect", pathname);
      return NextResponse.redirect(loginUrl);
    }

    // For other routes, continue
    return NextResponse.next();
  }
}

export const config = {
  matcher: [
    /*
     * Match all request paths except for the following:
     * - _next/static (static files)
     * - _next/image (image optimization files)
     * - favicon.ico (favicon file)
     * - api routes (API routes)
     */
    "/((?!_next/static|_next/image|favicon.ico|api).*)",
  ],
};
